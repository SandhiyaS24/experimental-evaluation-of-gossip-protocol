Start Zookeeper
bin/zookeeper-server-start.sh config/zookeeper.properties

Start Kafka Server
bin/kafka-server-start.sh config/server.properties

Create gossip topic
bin/kafka-topics.sh --create --bootstrap-server localhost:9092 --replication-factor 1 --partitions 1 --topic gossip

List of topics
bin/kafka-topics.sh --list --bootstrap-server localhost:9092

Starting a consumer for the topic gossip
bin/kafka-console-consumer.sh --bootstrap-server localhost:9092 --topic gossip --from-beginning

Produce a message to a topic
bin/kafka-console-producer.sh --broker-list localhost:9092 --topic gossip
